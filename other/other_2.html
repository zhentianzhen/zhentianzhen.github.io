logstash配置
使用redis作为broker：
input {
    redis {
        host => "10.10.0.1"
        port => "6379"
        key => "xxx"
        data_type => "list"
        type => "xxx"
    }
}

output {
    elasticsearch {
        hosts => ["10.10.0.2:80"]
        index => "xxx"
    }
}

多个input时:
input {
    redis {
        host => "10.10.0.1"
        port => "6379"
        key => "xxx"
        data_type => "list"
        type => "xxx"
    }
    redis {
        host => "10.10.0.1"
        port => "6379"
        key => "yyy"
        data_type => "list"
        type => "yyy"
    }
    redis {
        host => "10.10.0.1"
        port => "6379"
        key => "zzz"
        data_type => "list"
        type => "zzz"
    }
}

output {
    if [type] == "xxx" {
        elasticsearch {
            hosts => ["10.10.0.2:80"]
            index => "xxx"
        }
    }
    if [type] == "yyy" {
        elasticsearch {
            hosts => ["10.10.0.2:80"]
            index => "yyy"
        }
    }
    if [type] == "zzz" {
        elasticsearch {
            hosts => ["10.10.0.2:80"]
            index => "zzz"
        }
    }
}

存储nginx日志：
input {
    file {
        path => "/data/logs/host.access.log"
        start_position => beginning
    }
}

filter {
    grok {
        match => { "message" => "%{IPORHOST:client} (%{USER:ident}|-) (%{USER:auth}|-) \[%{HTTPDATE:timestamp}\] \"(?:%{WORD:verb} %{NOTSPACE:request}(?: HTTP/%{NUMBER:http_version})?|-)\" %{NUMBER:response} %{NUMBER:bytes} (%{QS:referrer}|-) (%{QS:agent}|-)"}
    }
}

output {
    elasticsearch {
        hosts => ["10.10.0.2:80"]
        index => "nginx"
    }
}

output { stdout {} }
