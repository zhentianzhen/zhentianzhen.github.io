protobuf的使用

      protobuf使用示例


      首先我们需要安装protobuf：


git clone https://github.com/google/protobuf.git
cd protobuf
./autogen.sh
./configure
make && make install

      然后我们定义一个消息结构：


package main;

enum MsgType {
  Normal       = 1;  
  Private      = 2;  
  System       = 3;  
  BroadCast    = 4;  
  Error        = 5;  
  Shield       = 6;  
  CancelShield = 7;  
  Offline      = 8;  
  Apply        = 9;  
  Status       = 10; 
  Invite       = 11; 
  Room         = 12; 
  Friend       = 13; 
  Custom       = 14; 
  Recent       = 15; 
  Corp         = 16; 
  Tips         = 17; 
}

message Msg {
  required MsgType type        = 1 [default=Normal]; 
  required string  content     = 2;                      
  optional string  from        = 3;                      
  optional string  from_name   = 4;                       
  optional string  to          = 5;                      
  optional string  to_name     = 6;                       
  optional string  time        = 7;                      
  optional string  color       = 8;                      
  optional string  from_avatar = 9;                       
  optional string  to_avatar   = 10;                      
  optional string  vip         = 11;                     
  optional string roominfo     = 12;                     
}

      执行如下语句，即可生成相应go文件：


protoc --go_out=. *.proto
