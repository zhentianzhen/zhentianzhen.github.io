golang读写excel文件

      golang读写excel文件示例：


package main

import (
  "fmt"
  "github.com/tealeg/xlsx"
  "strconv"
)

func main() {
  write()
  read()
}

func read() {
  fileName := "test.xlsx"
  xlFile, err := xlsx.OpenFile(fileName)
  if err != nil {
    panic(err)
  }

  for _, v := range xlFile.Sheets {
    for key, val := range v.Rows {
      fmt.Println(key)
      for _, item := range val.Cells {
        fmt.Println(item.String())
      }
    }
  }
}

func write() {
  var file *xlsx.File
  var sheet *xlsx.Sheet
  var row *xlsx.Row
  var cell *xlsx.Cell
  var err error

  file = xlsx.NewFile()
  for i := 0; i < 3; i++ {
    sheet, err = file.AddSheet("sheet" + strconv.Itoa(i))
    if err != nil {
      panic(err)
    }
    for i := 0; i < 4; i++ {
      row = sheet.AddRow()
      for j := 0; j < 5; j++ {
        cell = row.AddCell()
        cell.Value = "j" + strconv.Itoa(j)
      }
    }
  }

  err = file.Save("test.xlsx")
  if err != nil {
    panic(err)
  }
}
