MongoDB replset+auth配置
//首先生成keyfile
openssl rand -base64 753 > keyfile
chmod 600 keyfile 

//master配置
logpath=/data/mongo/replset_auth/master/mongod.log
logappend=true
fork=true
port=27017
dbpath=/data/mongo/replset_auth/master/data
pidfilepath=/data/mongo/replset_auth/master/mongod.pid
replSet=gama
storageEngine=wiredTiger
keyFile=/data/mongo/replset_auth/keyfile

//slave配置
logpath=/data/mongo/replset_auth/slave/mongod.log
logappend=true
fork=true
port=27018
dbpath=/data/mongo/replset_auth/slave/data
pidfilepath=/data/mongo/replset_auth/slave/mongod.pid
replSet=gama
storageEngine=wiredTiger
keyFile=/data/mongo/replset_auth/keyfile

//vote配置
logpath=/data/mongo/replset_auth/vote/mongod.log
logappend=true
fork=true
port=27019
dbpath=/data/mongo/replset_auth/vote/data
pidfilepath=/data/mongo/replset_auth/vote/mongod.pid
replSet=gama
storageEngine=wiredTiger
keyFile=/data/mongo/replset_auth/keyfile

///////////////////////////////////
mongod -f master.conf
mongod -f slave.conf
mongod -f vote.conf

//创建replset
mongo
>config = {_id: "gama", members: [{_id: 0, host: "127.0.0.1:27017"}, {_id:1, host: "127.0.0.1:27018"}]}
>rs.initiate(config)

//创建用户
>use admin
>db.createUser({user: "admin", pwd: "123456", roles: ["userAdminAnyDatabase", "dbAdminAnyDatabase", "readWriteAnyDatabase", "clusterAdmin"]})

//添加投票节点
>rs.addArb("127.0.0.1:27019")
>rs.status()
